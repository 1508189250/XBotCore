137 void XBot::XBotXDDP::update()
138 {
139  // Motor
140  for( auto& f: fd_read) {
141 
142  // write to the NRT publisher to command the RobotStateTX in the pdo_motor buffer
143  XBot::RobotState::pdo_tx actual_pdo_tx = pdo_motor.at(f.first)->RobotStateTX;
144  fd_write.at(f.first).write(actual_pdo_tx);
145 
146  // NOTE the single joint element can only be controlled by either the RT or the N-RT!
147 
148  // reading from the NRT subscriber pipes to update the RobotStateRX in the pdo_motor buffer
149  fd_read.at(f.first).read(*pdo_motor.at(f.first));
150 
151  }
152 }
