209 void XBot::CommunicationHandler::th_loop(void*)
210 {
211  /* Receive Master Communication Interface to switch NRT framework at runtime */
212 
213  for(auto comm_ifc : _communication_ifc_vector) {
214 
215  std::string master;
216 
217  if( comm_ifc->receiveMasterCommunicationInterface(master) ) {
218 
219  if ( master == "ROS" ||
220  master == "ros"
221  ) {
222  std::cout << "Switching to ROS Master Communication Interface" << std::endl;
223 
224 #ifdef USE_ROS_COMMUNICATION_INTERFACE
225  _master_communication_ifc = _ros_communication;
226  // HACK restarting XBotCommunicationPlugin
227  std::string cmd = "stop";
228  _switch_pub_vector[xbot_communication_idx].write(cmd);
229  sleep(1);
230  cmd = "start";
231  _switch_pub_vector[xbot_communication_idx].write(cmd);
232 #else
233  std::cerr << "ERROR: ROS Master Communication Interface not compiled" << std::endl;
234 #endif
235  }
236 
237  else if ( master == "YARP" ||
238  master == "yarp"
239  ) {
240  std::cout << "Switching to YARP Master Communication Interface" << std::endl;
241 
242 #ifdef USE_YARP_COMMUNICATION_INTERFACE
243  _master_communication_ifc = _yarp_communication;
244  // HACK restarting XBotCommunicationPlugin
245  std::string cmd = "stop";
246  _switch_pub_vector[xbot_communication_idx].write(cmd);
247  sleep(1);
248  cmd = "start";
249  _switch_pub_vector[xbot_communication_idx].write(cmd);
250 #else
251  std::cerr << "ERROR: YARP Master Communication Interface not compiled" << std::endl;
252 #endif
253  }
254  }
255  }
256 
257  /* Receive commands on switch ports */
258  for(auto comm_ifc : _communication_ifc_vector) {
259  for(int i = 0; i < _plugin_names.size(); i++){
260  std::string command;
261  if( comm_ifc->receiveFromSwitch(_switch_names[i], command) ){
262  _switch_pub_vector[i].write(command);
263  }
264  if( comm_ifc->receiveFromCmd(_command_names[i], command) ){
265  _command_pub_vector[i].write(command);
266  }
267 
268  XBot::Command status;
269  if( _status_sub_vector[i].read(status) ){
270  comm_ifc->setPluginStatus(_plugin_names[i], status.str());
271  }
272  }
273  }
274 
275  /* Update XDDP */
276  _xddp_handler->update();
277 
278  /* Read robot state from RT layer and update robot */
279  _robot->sense(false);
280 
281  /* Log */
282  _robot->log(_logger, double(XBot::get_time_ns()) / 1e9);
283 
284  /* Publish robot state to all frameworks */
285  for(auto comm_ifc : _communication_ifc_vector){
286  comm_ifc->sendRobotState();
287  }
288 
289  /* Receive commands from the master communication handler,
290  * i.e. the only one enabled to send commands to the robot */
291  _master_communication_ifc->receiveReference(); // this updates robot
292 
293  /* Run external plugins */
294  for( XBot::IOPlugin * io_plugin_ptr : _io_plugin_ptr ){
295  if(io_plugin_ptr) io_plugin_ptr->run();
296  }
297 
298  /* Send received commands to the RT layer */
299  _robot->move();
300 
301 }
